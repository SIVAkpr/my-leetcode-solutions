1class Solution {
2    public int maxProduct(int n) {

4        String digits=String.valueOf(n);
5        for(int i=0;i<digits.length();i++) {
6            for(int j=i+1;j<digits.length();j++) {
7                int d1=digits.charAt(i)-'0';

11            }
12        }
13        
14        return maxpro;
15    }
